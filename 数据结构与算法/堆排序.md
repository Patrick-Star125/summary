# 基础代码

~~~c++
void adjust(vector<int>& arr, int len, int index){
    leftIndex = index*2+1;
    rightIndex = index*2+2;
    maxIndex = index;
    if(leftIndex<len && arr[maxIndex]<arr[leftIndex]) maxIndex=leftIndex;
    if(rightIndex<len && arr[maxIndex]<arr[rightIndex]) maxIndex=rightIndex;
    if(maxIndex!=index){
        swap(arr[maxIndex], arr[index]);
        adjust(arr, len, maxIndex);
    }
}

void headSort(vector<int>& arr){
    int size = arr.size();
    for(int i=size/2-1;i>=0;i--){
        adjust(arr, i);
    }
    for(int i = size - 1; i >= 1; i--){
        swap(arr[0], arr[i]);
        adjust(arr, i, 0);
    }
}
~~~

堆排序的基础步骤是按照索引自上而下的调整堆，一次调整保证当前索引节点树符合堆要求
